Class Timezone

Timezone representation.

Example

var vcalendar;
var timezoneComp = vcalendar.getFirstSubcomponent('vtimezone');
var tzid = timezoneComp.getFirstPropertyValue('tzid');

var timezone = new ICAL.Timezone({
  component: timezoneComp,
  tzid
});
